<ALPHA> ::= A-Za-z

<WORD> ::= 'A-Za-z_/.*$'

<FILE> ::= <WORD>

<DIRECTION> ::= ">"
			 | "<" 
			 | ">>" 
			 | "<<"

<REDIRECTION> ::= <DIRECTION> <FILE>

<ARGS> ::= <WORD> <ARGS>
		| <WORD>

<COMMAND> ::= [<REDIRECTION>] <EXEC> [<ARGS>]
			| <EXEC> [<REDIRECTION> <ARGS>]
			| <EXEC> [<ARGS> <REDIRECTION>]

<PIPELINE> ::= <COMMAND> "|" <PIPELINE>
			|  <COMMAND>
			| "(" <COMMAND> ")"

<COMMANDLINE> ::= <PIPELINE> "&&" <COMMANDLINE>
			   |  <PIPELINE> "||" <COMMANDLINE>
			   |  <PIPELINE>
