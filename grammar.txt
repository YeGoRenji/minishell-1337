<ALPHA> ::= A-Za-z

<PATH> ::= <ALPHA>

<WORD> ::= 'A-Za-z_/.*$'

<FILE> ::= <WORD>

<DIRECTION> ::= ">" | "<" | ">>" | "<<"

<REDIRECTION> ::= <DIRECTION> <FILE>

<ARGS> ::= <WORD> <ARGS> | <WORD>

// This means that $"" is not really an env (dont expand)
<ENV> ::= $<ALPHA>+

<COMMANDLINE> ::= [<REDIRECTION>] <EXEC> [<ARGS>]
				| <EXEC> [<REDIRECTION> <ARGS>]
				| <EXEC> [<ARGS> <REDIRECTION>]

<PIPELINE> ::= <COMMAND> '|' <PIPELINE>
			|  <COMMANDLINE>
