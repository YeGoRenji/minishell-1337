<ALPHA> ::= A-Za-z

<WORD> ::= 'A-Za-z_/.*$'

<FILE> ::= <WORD>

<DIRECTION> ::= ">"
			 | "<" 
			 | ">>" 
			 | "<<"

<REDIRECTION> ::= <DIRECTION> <FILE>

<ARGS> ::= <WORD> <ARGS>
		| <WORD>

<REDIR> ::= [<REDIRECTION>] <EXEC> [<ARGS>]
			| <EXEC> [<REDIRECTION> <ARGS>]
			| <EXEC> [<ARGS> <REDIRECTION>]

<PIPELINE> ::= <REDIR> "|" <PIPELINE>
			|  <REDIR>
			| "(" <REDIR> ")"

<COMMANDLINE> ::= <COMMANDLINE> "&&" <COMMANDLINE>
			   |  <COMMANDLINE> "||" <COMMANDLINE>
			   |  <PIPELINE>
